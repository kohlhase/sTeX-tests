\begin{mhmodnl}{FOO}{de}
  \begin{definition}
    Ein deutsches Modulbindung das zwei Symbole definiert: eine Konstante $\FOO$ und eine
    Funktion $\foobar{X}$. Wir k"onnen auch "uber verschiedene Konzepte reden:
    \begin{itemize}
    \item   \trefi[?foo]{derfoo}
    \item   \trefii[?foo-bar]{derfoo}{bar}
    \item   \trefiii[?foo-bar-baz]{derfoo}{bar}{baz}
    \item   \trefiv[?foo-bar-baz-bam]{derfoo}{bar}{baz}{bam}
    \end{itemize}
  \end{definition}
\end{mhmodnl}
